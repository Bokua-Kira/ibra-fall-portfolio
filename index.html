class ReapprovisionnementOptimise:
    def __init__(self, df_ventes, stock_actuel):
        self.df_ventes = df_ventes
        self.stock = stock_actuel
        self.seuil_rupture = 10  # Stock minimum avant r√©appro
    
    def calculer_ventes_moyennes(self, produit, periode=7):
        """Calcule la moyenne des ventes sur X jours"""
        ventes_produit = self.df_ventes[
            self.df_ventes['produit'] == produit
        ]
        return ventes_produit['quantite'].tail(periode).mean()
    
    def predire_rupture(self, produit):
        """Pr√©dit si un produit va √™tre en rupture"""
        stock_actuel = self.stock.get(produit, 0)
        ventes_journalieres = self.calculer_ventes_moyennes(produit)
        
        if ventes_journalieres > 0:
            jours_restants = stock_actuel / ventes_journalieres
            return jours_restants < 3  # Alerte si moins de 3 jours
        return False
    
    def commander(self, produit):
        """Calcule la quantit√© √† commander"""
        ventes_7j = self.calculer_ventes_moyennes(produit, 7)
        ventes_30j = self.calculer_ventes_moyennes(produit, 30)
        
        # Poids plus important sur les ventes r√©centes
        ventes_prevues = (ventes_7j * 0.7) + (ventes_30j * 0.3)
        
        # Commander pour 15 jours + stock de s√©curit√©
        quantite = (ventes_prevues * 15) * 1.2  # 20% de marge
        
        return int(quantite)
